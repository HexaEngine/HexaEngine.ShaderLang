namespace Test;
using Core;

SamplerState samplerState;
Texture2D fontTex;

struct MyStruct
{
    float4 pos;

    public MyStruct(float4 pos)
    {
        this.pos = pos;
    }

    public void Add(float4 p)
    {
        pos += p;
    }

    public static implicit operator float4(in MyStruct val)
    {
        return val.pos;
    }
}

float4 main(PS_INPUT pixel) : SV_TARGET
{
    float array[10];
    float4 color = pixel.color * fontTex.Sample(samplerState, pixel.tex);
    for (int i = 0; i < 10; i++)
    {
        array[i] = i + 1f;
    }
    return color;
}
